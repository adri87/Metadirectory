  170.     * This function combines the ranking functions specified in
  171.     * {@link romkingComponentSet::$aRankingMethod $aRankingMethod}
  172.     * and calculates the overall ranking score for each component
  173.     * in the current set.
  174.     *
  175.     * @uses romkingComponentSet::$aRankingMethod
  176.     * @uses romkingComponentSet::$aSet
  177.     * @return void 
  178.     */
  179.     private function calculateRankingScores () {
  180.  
  181.         # sums for all ranking scores
  182.         # needed to normalize the scores
  183.         $aSums = array();
  184.  
  185.         # iterate over the whole set
  186.         foreach( $this->aSet as $iIndex => $aComp )
  187.             # iterate over ranking functions
  188.             foreach( $this->aRankingMethod as $aRankingFunction )
  189.                 # test if ranking function is relevant and exists
  190.                 if( method_exists( $this, "componentRankingFunction_" . $aRankingFunction[ 1 ] ) &&
  191.                     $aRankingFunction[ 0 ] > 0 ) {
  192.  
  193.                     # calculate score of current component in current ranking function
  194.                     $this->aSet[ $iIndex ][ 'score' ][ $aRankingFunction[ 1 ] ] = call_user_func( array( $this, "componentRankingFunction_" . $aRankingFunction[ 1 ] ), $aComp[ 'obj' ] );
  195.  
  196.                     # add the score to the overall sum of this ranking function
  197.                     $aSums[ $aRankingFunction[ 1 ] ] += $this->aSet[ $iIndex ][ 'score' ][ $aRankingFunction[ 1 ] ];
  198.  
  199.                 }
  200.  
  201.         # normalize each score
  202.         foreach( $this->aSet as $iIndex => $aComp ) 
  203.             foreach( $aComp[ 'score' ] as $sScoringFunction => $iScore )
  204.                 $this->aSet[ $iIndex ][ 'score' ][ $sScoringFunction ] = $iScore / $aSums[ $sScoringFunction ];
  205.  
  206.         # calculate overall score according to the
  207.         # weight factors of the specific ranking functions
  208.         foreach( $this->aSet as $iIndex => $aComp )  {
  209.  
  210.             # init overall score
  211.             $iOverallScore = 0;
  212.  
  213.             # init sum of weights
  214.             $iWeightSum = 0;
  215.  
  216.             foreach( $this->aRankingMethod as $aRankingFunction ) {
  217.                 $iOverallScore += $aRankingFunction[ 0 ] * $aComp[ 'score' ][ $aRankingFunction [ 1 ] ];
  218.                 $iWeightSum += $aRankingFunction[ 0 ];
  219.             }
  220.  
  221.             # store normalized overall score
  222.             $this->aSet[ $iIndex ][ 'score' ][ 'overall' ] = $iOverallScore / $iWeightSum;
  223.  
  224.         }
  225.         
  226.  
  227.     }

  243.     public static function compareComponentsAccordingToScore ( $aCompA, $aCompB ) {
  244.         
  245.         # returns a value greater than, equal to or lower than 0
  246.         # if the score of component A is greater than, equal to or lower than
  247.         # the score of component B
  248.         # NOTE: scores are multiplied with high int in order to solve problem
  249.         #    with the precision on float
  250.         return ( round( $aCompA[ 'score' ][ 'overall' ] * 1000000000 ) - round( $aCompB[ 'score' ][ 'overall' ] * 1000000000 ) ) * ( -1 );
  251.  
  252.     }

  429.             array_push( $this->aGraph[ $this->aMashups[ $aResult[ 'mashupIRI' ] ] ], $this->aAPIs[ $aResult[ 'apiIRI'] ] );

  434.     /**
  435.     * Calculation of Betweenness and Closeness Centralities
  436.     *
  437.     * This is the implementation of
  438.     * {@link http://www.informatik.uni-konstanz.de/algo/publications/b-fabc-01.pdf}
  439.     * Brandes' Algorithm} to calculate the Betweenness Centrality
  440.     * as well as the Closeness Centrality in the Mashup-API-Graph.
  441.     * Closeness Centrality is modified to be applicable on disconnected
  442.     * graphs to according to
  443.     * {@link http://toreopsahl.com/2010/03/20/closeness-centrality-in-networks-with-disconnected-components/}
  444.     * Tore Opsahl's Proposal}
  445.     *
  446.     * @uses romkingRanking::$aBetweennessCentralities
  447.     * @uses romkingRanking::$aClosenessCentralities
  448.     * @return array 
  449.     */
  450.     public function calculateBetweennessAndClosenessCentralities ( ) {
  451.  
  452. //         $time = microtime();
  453. //         $time = explode(' ', $time);
  454. //         $time = $time[1] + $time[0];
  455. //         $begintime = $time;
  456.  
  457.         # Needed for writing to the triple store
  458.         $aIRIs = array_flip( $this->aAPIs );
  459.  
  460.         # union of Mashups and APIs
  461.         # i.e. the set of vertices of the graph
  462.         foreach( $this->aGraph as $iIndex => $aNeighbours )
  463.             if( $aNeighbours )
  464.                 $aBigV[ $iIndex ] = $aNeighbours;
  465.  
  466.         # amount of vertices in the graph
  467.         $iAmountOfVertices = sizeof( $aBigV );
  468.  
  469.         # CB[v] = 0, v€V
  470.         $this->aBetweennessCentralities = array();
  471.  
  472.         # CC[v] = 0, v€V
  473.         $this->aClosenessCentralities = array();
  474.  
  475.         # for s € V
  476.         foreach ( $aBigV as $iSIndex => $aNeighboursOfS ) {
  477.  
  478.             # S <- empty stack
  479.             $aS = array();
  480.  
  481.             # P[w] <- empty list, w € V
  482.             $aP = array();
  483.             foreach( $aBigV as $iIndex => $aNeighbours ) // for( $iIndex = 0; $iIndex < $iAmountOfVertices; $iIndex++ )
  484.                 $aP[ $iIndex ] = array();
  485.  
  486.             # sigma(t) = 0, t € V; sigma(s) = 1
  487.             $aSigma = array( $iSIndex => 1 );
  488.  
  489.             # d(t) = -1, t € V; d(s) = 0
  490.             $aD = array();
  491.             foreach( $aBigV as $iIndex => $aNeighbours ) // for( $iIndex = 0; $iIndex < $iAmountOfVertices; $iIndex++ )
  492.                 $aD[ $iIndex ] = -1;
  493.             $aD[ $iSIndex ] = 0;
  494.  
  495.             # Q <- empty queue
  496.             $aQ = array();
  497.  
  498.             # enqueue s -> Q
  499.             array_push( $aQ, $iSIndex );
  500.  
  501.             # while (Q not empty) do
  502.             while ( sizeof ( $aQ ) ) {
  503.  
  504.                 # unqueue v <- Q
  505.                 $iV = array_shift( $aQ );
  506.  
  507.                 # push v -> S
  508.                 array_push( $aS, $iV );
  509.  
  510.                 # foreach neighbour w of v do
  511.                 foreach( $aBigV[ $iV ] as $iW ) {
  512.  
  513.                     # // w found for the first time?
  514.                     # if d(w) < 0 then
  515.                     if( $aD[ $iW ] < 0 ) {
  516.  
  517.                         # enqueue w -> Q
  518.                         array_push( $aQ, $iW );
  519.  
  520.                         # d(w) <- d(v) + 1
  521.                         $aD[ $iW ] = $aD[ $iV ] + 1;
  522.  
  523.                     }
  524.  
  525.                     # // shortest path to w via v?
  526.                     # if d(w) = d(v) + 1 then
  527.                     if( $aD[ $iW ] == ( $aD[ $iV ] + 1 ) ) {
  528.  
  529.                         # sigma(w) <- sigma(w) + sigma(v)
  530.                         $aSigma[ $iW ] = $aSigma[ $iW ] + $aSigma[ $iV ];
  531.  
  532.                         # append v -> P(w)
  533.                         array_push( $aP[ $iW ], $iV );
  534.  
  535.                     }
  536.                 }
  537.             }
  538.  
  539.             # delta(v) <- 0, v € V
  540.             $aDelta = array();
  541.  
  542.             # // S returns vertices in order of non-increasing distance from s
  543.             # while (S not empty) do
  544.             while ( sizeof( $aS ) ) {
  545.  
  546.                 # pop w <- S
  547.                 $iW = array_pop( $aS );
  548.  
  549.                 # for v € P(w) do delta(v) <- delta(v) + (sigma(v) / sigma(w)) * (1 + delta(w))
  550.                 foreach( $aP[ $iW ] as $iV )
  551.                     $aDelta[ $iV ] = $aDelta[ $iV ] + ( $aSigma[ $iV ] / $aSigma[ $iW ] ) * ( 1 + $aDelta[ $iW ] );
  552.     
  553.                 # if w != s then CB(w) <- CB(w) + delta(w)
  554.                 if( $iW != $iSIndex )
  555.                     $this->aBetweennessCentralities[ $iW ] += $aDelta[ $iW ];
  556.             }
  557.  
  558.             // Closeness Centrality
  559.             $this->aClosenessCentralities[ $iSIndex ]  = array_sum( array_map( function ( $v ) { return ( $v > 0 ) ? ( 1 / $v ) : 0; }, $aD ) );
  560.  
  561. //             $i++;
  562. //             echo $i . " of " . $iAmountOfVertices . "<br />";
  563. //             if( $i == 500 ) {
  564. //                 $time = microtime();
  565. //                 $time = explode(" ", $time);
  566. //                 $time = $time[1] + $time[0];
  567. //                 $endtime = $time;
  568. //                 $totaltime = ($endtime - $begintime);
  569. //                 echo $totaltime. ' seconds.';
  570. // 
  571. //                 exit;
  572. //             }
  573.         }
  574.     }

  692.     /**
  693.     * Calculation of Eigenvector Centralities
  694.     *
  695.     * Power iteration method
  696.     *
  697.     * @uses romkingRanking::$aEigenvectorCentralities
  698.     * @return array 
  699.     */
  700.     private function calculateEigenvectorCentralities ( ) {
  701.  
  702. //         $time = microtime();
  703. //         $time = explode(' ', $time);
  704. //         $time = $time[1] + $time[0];
  705. //         $begintime = $time;
  706.  
  707.         # Number of iterations
  708.         $iAmountOfIterations = 30;
  709.  
  710.         # union of Mashups and APIs
  711.         # i.e. the set of vertices of the graph
  712.         # NOTE: For this we will need the transposed
  713.         # Matrix for dealing with directed graphs
  714.         $aBigV = array();
  715.         foreach( $this->aGraph as $iIndex => $aNeighbours ) {
  716.             $aBigV[ $iIndex ] = array();
  717.             $iAmountOfNeighbours = sizeof( $aNeighbours );
  718.             foreach( $aNeighbours as $iNeighbourIndex => $iNeighbour )
  719.                 $aBigV[ $iIndex ][ $iNeighbourIndex ] = array( 
  720.                     'dir' => $iNeighbour,
  721.                     'weight' => 1 / $iAmountOfNeighbours
  722.                 );
  723.         }
  724.  
  725.         # amount of vertices in the graph
  726.         $iAmountOfVertices = sizeof( $aBigV );
  727.  
  728.         # CE[v] = 0, v€V
  729.         $this->aEigenvectorCentralities = array();
  730.  
  731.         # b(0)
  732.         $aB = array( 0 => array() );
  733.         foreach( $aBigV as $iIndex => $aNeighbours )
  734.             $aB[ 0 ][ $iIndex ] = 1 / $iAmountOfVertices; // 0;
  735.  
  736.         //$aB[0][0] = 1;
  737.  
  738.         # Power iteration
  739.         for( $i = 1; $i <= $iAmountOfIterations; $i++ ) {
  740.  
  741.             foreach( $aBigV as $iIndex => $aNeighbours )
  742.                 foreach( $aNeighbours as $aNeighbour )
  743.                     $aB[ $i ][ $aNeighbour[ 'dir' ] ] += $aNeighbour[ 'weight' ] * $aB[ ( $i - 1 ) ][ $iIndex ];
  744.  
  745.             $iNormFactor = array_sum( $aB[ $i ] );
  746.             
  747. //             foreach( $aB [ $i ] as $iIndex => $iValue )
  748. //                 $aB[ $i ][ $iIndex ] = $iValue / $iNormFactor;
  749.  
  750.             # echo $i . ". iteration<br />";
  751.         }
  752.  
  753. //         $time = microtime();
  754. //         $time = explode(" ", $time);
  755. //         $time = $time[1] + $time[0];
  756. //         $endtime = $time;
  757. //         $totaltime = ($endtime - $begintime);
  758. //         echo $totaltime. ' seconds needed for ' . $iAmountOfIterations . ' iterations.';
  759.  
  760.         $this->aEigenvectorCentralities = $aB[ $iAmountOfIterations ];
  761.     }

  812.
