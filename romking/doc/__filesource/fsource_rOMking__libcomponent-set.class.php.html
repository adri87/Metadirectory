  154.     /**
  155.     * User Query analysis
  156.     *
  157.     * This function analyzes the user's input and parses it.
  158.     * The results will be stored as an array in
  159.     * {@link romkingComponentSet::$aUserQuery $aUserQuery}.
  160.     *
  161.     * @uses romkingComponentSet::$sQueryExpression
  162.     * @uses romkingComponentSet::$aUserQuery
  163.     *
  164.     * @return array 
  165.     */
  166.     protected function analyzeUserInputString () {
  167.  
  168.         # split query string on white space
  169.         $aUserQuery = explode( " ", $this->sQueryExpression );// str_getcsv( $this->sQueryExpression, " " ); //
  170.  
  171.         # look for operators
  172.         # note: the constructor has strtolower()ed the string
  173.         for( $i = 0; $i < sizeof( $aUserQuery ); $i++ )
  174.             if (    $aUserQuery[ $i ] == "or" ||
  175.                 $aUserQuery[ $i ] == "and" )
  176.                     $this->aUserQuery[ $i ] = array(
  177.                                     "operator" => strtoupper( $aUserQuery[ $i++ ] ),
  178.                                     "term" => $aUserQuery[ $i ],
  179.                                     "type" => null
  180.                                 );
  181.             else
  182.                 $this->aUserQuery[ $i ] = array(
  183.                                 "operator" => $this->sDefaultSearchQueryOperator,
  184.                                 "term" => $aUserQuery[ $i ],
  185.                                 "type" => null
  186.                             );
  187.  
  188.         # look for type definitions in the search terms
  189.         foreach( $this->aUserQuery as $iIndex => $aValue )
  190.             if( preg_match ( "#:#", $aValue[ 'term' ] ) )
  191.                 list( $this->aUserQuery[ $iIndex ][ 'type' ], $this->aUserQuery[ $iIndex ][ 'term' ] ) = split( ":", $aValue[ 'term' ] );
  192.  
  193.         # return resulting array
  194.         return $this->aUserQuery;
  195.  
  196.     }

  202.     /**
  203.     * Build SPARQL Query
  204.     *
  205.     * This function is the core of the matchmaking process. It
  206.     * defines the SPARQL query to fetch all of the relevant components
  207.     * from the triple store based on the user's query
  208.     *
  209.     * @uses romkingComponentSet::$aMatchingMethod
  210.     * @uses romkingComponentSet::$aUserQuery
  211.     * @uses romkingComponentSet::$sSPARQLQuery
  212.     * @return string 
  213.     */
  214.     protected function buildSPARQLQuery () {
  215.  
  216.         # initialize the query
  217.         # - defining namespaces
  218.         # - selecting the identifier (IRI) as well as
  219.         # - the degree-centrality of the component
  220.         $this->sSPARQLQuery  = "\r\n";
  221.         $this->sSPARQLQuery .= "PREFIX  rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>\r\n";
  222.         $this->sSPARQLQuery .= "PREFIX  rdfs: <http://www.w3.org/2000/01/rdf-schema#>\r\n";
  223.         $this->sSPARQLQuery .= "PREFIX  oml: <http://www.ict-omelette.eu/schema.rdf#>\r\n";
  224.         $this->sSPARQLQuery .= "PREFIX  omltags: <http://www.ict-omelette.eu/omr/tags/>\r\n";
  225.         $this->sSPARQLQuery .= "PREFIX  ctag: <http://commontag.org/ns#>\r\n";
  226.         $this->sSPARQLQuery .= "PREFIX  dc: <http://purl.org/dc/elements/1.1/>\r\n";
  227.         $this->sSPARQLQuery .= "SELECT ?IRI\r\n"; // (count(distinct ?mashup) as ?degreeCentrality)
  228.         $this->sSPARQLQuery .= "\tWHERE {\r\n";
  229.         $this->sSPARQLQuery .= "\t\t?api oml:api ?IRI .\r\n";
  230.         $this->sSPARQLQuery .= "\t\t?api rdf:type oml:Service .\r\n";
  231.         $this->sSPARQLQuery .= "\t\t?api oml:dataFormat ?dataFormat .\r\n";
  232.         $this->sSPARQLQuery .= "\t\t?api ctag:tagged ?apiTag .\r\n";
  233.         $this->sSPARQLQuery .= "\t\t?api rdfs:label ?apiTitle .\r\n";
  234.         $this->sSPARQLQuery .= "\t\t?api dc:description ?apiDescription .\r\n";
  235.         $this->sSPARQLQuery .= "\t\t?mashup oml:uses ?api .\r\n";
  236.         $this->sSPARQLQuery .= "\t\t?mashup ctag:tagged ?mashupTag .\r\n";
  237.         
  238.         # initializing OR-Filters and AND-Filters
  239.         $aFilters = array(
  240.                     "OR" => array (),
  241.                     "AND" => array ()
  242.                 );
  243.  
  244.         # handling every search term in the user query
  245.         foreach( $this->aUserQuery as $iIndex => $aQueryTerm ) {
  246.             
  247.             # add respective operator of current term to
  248.             # the query string
  249.             # $this->sSPARQLQuery .= " " . $aQueryTerm[ 'operator' ] . " ";
  250.  
  251.             if( $aQueryTerm[ 'type' ] )
  252.                 # the current term has a specific type, e.g. dataformat
  253.                 # connected to it. the following block will handle those
  254.                 # specific options
  255.                 switch( $aQueryTerm[ 'type' ] ) {
  256.                     default:
  257.                     break; case "dataformat":
  258.                         $this->sSPARQLQuery .= "\t\tFILTER regex( ?dataFormat, \"" . $aQueryTerm[ 'term' ] . "\", \"i\" ) .\r\n";
  259.                     break;
  260.                 }
  261.             else {
  262.  
  263.                 # "if" for every aspect of the general information
  264.                 # about the component that should be considered
  265.  
  266.                 # auxiliary array for the necessary conditions
  267.                 $aCurrentConditions = array();
  268.  
  269.                 # Tags of the API
  270.                 if( in_array( "api-tags", $this->aMatchingMethod ) )
  271.                     $aCurrentConditions[] = "?apiTag = omltags:" . $aQueryTerm[ 'term' ];
  272.  
  273.                 # Tags of mashups using the API
  274.                 if( in_array( "mashup-tags", $this->aMatchingMethod ) )
  275.                     $aCurrentConditions[] = "?mashupTag = omltags:" . $aQueryTerm[ 'term' ];
  276.  
  277.                 # Title of the API
  278.                 if( in_array( "api-name", $this->aMatchingMethod ) )
  279.                     $aCurrentConditions[] = "regex( ?apiTitle, \" " . $aQueryTerm[ 'term' ] . " \", \"i\" )";
  280.  
  281.                 # Description of the API
  282.                 if( in_array( "api-description", $this->aMatchingMethod ) )
  283.                     $aCurrentConditions[] = "regex( ?apiDescription, \" " . $aQueryTerm[ 'term' ] . " \", \"i\" )";
  284.  
  285.                 # Glueing Conditions
  286.                 $aFilters[ $aQueryTerm[ 'operator' ] ][] = implode( " || ", $aCurrentConditions );
  287.  
  288.             }
  289.         }
  290.  
  291.         # get out the deadpool tagged APIs
  292.         $this->sSPARQLQuery .=  "\t\tFILTER NOT EXISTS { ?api ctag:tagged omltags:deadpool }\r\n";#"\t\tFILTER (\r\n\t\t\t?apiTag != omltags:deadpool\r\n\t\t\t).\r\n";
  293.  
  294.         # concatenation of the filter expressions
  295.         if( $aFilters[ 'OR' ] )
  296.             $this->sSPARQLQuery .= "\t\tFILTER (\r\n\t\t\t( " . implode( " ) ||\r\n\t\t\t( ", $aFilters[ 'OR' ] ) ." )\r\n\t\t) .\r\n";
  297.         if( $aFilters[ 'AND' ] )
  298.             $this->sSPARQLQuery .= "\t\tFILTER (\r\n\t\t\t( " . implode( " ) &&\r\n\t\t\t( ", $aFilters[ 'AND' ] ) ." )\r\n\t\t) .\r\n";
  299.  
  300.         # finalize the query
  301.         # Grouping by the component's identifier
  302.         # optional limit
  303.         $this->sSPARQLQuery .= "\t}\r\nGROUP BY ?IRI";
  304.         if( $this->iSetSize )
  305.             $this->sSPARQLQuery .= " LIMIT " . $this->iSetSize; # ORDER BY DESC(?score)";
  306.         
  307.  
  308.     }
  309.  
  310.     /**
  311.     * Execute SPARQL Query
  312.     *
  313.     * This function sends the query to the triple store and
  314.     * receives the results which it handles afterwards. It
  315.     * returns {@link romkingComponentSet::$aSet $aSet}.
  316.     *
  317.     * @uses romkingComponentSet::$oTripleStore
  318.     * @uses romkingComponentSet::$sSPARQLQuery
  319.     * @return array 
  320.     */
  321.     protected function executeSPARQLQuery () {
  322.  
  323.         # fetch result
  324.         $aRows = $this->oTripleStore->query( $this->sSPARQLQuery );
  325.         $aRows = $aRows->getRows();
  326.  
  327.         # store current set size
  328.         $this->iRealSetSize = sizeof( $aRows );
  329.         
  330.         # handle found data
  331.         foreach( $aRows as $iKey => $aResult ) {
  332.             $this->aSet[ $iKey ][ 'obj' ] = new romkingComponent( $aResult[ 'IRI' ] );
  333.             $this->aSet[ $iKey ][ 'score' ] = array();
  334.         }
  335.  
  336.     }
  337.

  343.     * Magic Method __toString()

  345.     * Will return

  352.         
  353.         $aReturn = $this->aSet;
  354.         foreach( $this->aSet as $iKey => $aValue ) 
  355.             $aReturn[ $iKey ][ 'obj' ] = json_decode( (string) $this->aSet[ $iKey ][ 'obj' ] );
  356.         
  357.         return json_encode( $aReturn );
  358.     }
